<script>
(function(){
  function initAuditSignalAccordion1932(){
    var root = document.getElementById("ce-audit-signal-accordion-1932");
    if(!root || root.getAttribute("data-audit-accordion-ready") === "true") return;
    root.setAttribute("data-audit-accordion-ready", "true");

    function togglePanel(panel){
      if(!panel) return;
      var isOpen = panel.classList.toggle("is-open");
      var trigger = panel.querySelector("[data-signal-trigger]");
      if(trigger){
        trigger.setAttribute("aria-expanded", isOpen ? "true" : "false");
      }
    }

    function activate(event){
      var trigger = event.target.closest("[data-signal-trigger]");
      if(!trigger || !root.contains(trigger)) return;

      event.preventDefault();
      event.stopPropagation();
      togglePanel(trigger.closest(".cex-panel"));
    }

    root.addEventListener("click", activate, true);
    root.addEventListener("keydown", function(event){
      if(event.key !== "Enter" && event.key !== " ") return;
      activate(event);
    }, true);
  }

  if(document.readyState === "loading"){
    document.addEventListener("DOMContentLoaded", initAuditSignalAccordion1932);
  } else {
    initAuditSignalAccordion1932();
  }

  window.addEventListener("load", initAuditSignalAccordion1932);
  setTimeout(initAuditSignalAccordion1932, 500);
  setTimeout(initAuditSignalAccordion1932, 1500);
})();
</script>
