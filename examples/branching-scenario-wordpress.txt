<script>
(function(){
  function initBranchingTree1932(){
    var root = document.getElementById("ce-branching-tree-1932");
    if(!root || root.getAttribute("data-branch-ready") === "true") return;
    root.setAttribute("data-branch-ready", "true");

    var nodes = {
      start: {
        step: 1,
        label: "First clinical move",
        question: "Where would you begin the assessment?",
        context: "The client has named anxiety and depression, but the scenario also includes identity-based stress, possible family-system risk, and privacy concerns.",
        choices: [
          {
            label: "Broaden the lens",
            text: "Ask about identity-based stress, safety at home, privacy, and support.",

          },
          {
            label: "Start narrow",

          },
          {

            text: "Ask what name, pronouns, and identity language they want used here and in documentation.",

          }
        ]
      },
      missed: {
        step: 2,
        label: "Course correction",
        question: "The client says, \"It is easier if I just stay quiet.\" What should you do next?",
        context: "This may signal self-silencing, fear of consequences, or a lack of safety in the family system.",
        choices: [
          {
            label: "Widen assessment",
            text: "Ask what happens when they speak up at home and whether they feel safe.",
            next: "risk",
            feedback: "<strong>Better direction.</strong> You are moving from a narrow symptom lens toward the relational and safety context."
          },
          {
            label: "Normalize it",
            text: "Tell them many young adults avoid family conflict and continue with anxiety treatment.",
            next: "end_minimize",
            feedback: "<strong>Careful.</strong> This risks minimizing identity-based harm. Avoidance may be protective, but it can also indicate danger, isolation, or loss of support."
          },
          {
            label: "Ask privacy first",
            text: "Clarify what the parent can see in the portal, billing, and documentation.",
            next: "risk",
            feedback: "<strong>Important move.</strong> If a parent pays for therapy, privacy and documentation can directly affect safety."
          }
        ]
      },
      privacy: {
        step: 2,
        label: "Confidentiality branch",
        question: "The parent pays for therapy. What is the next clinical priority?",
        context: "The client is unsure what information is visible in the portal. Privacy may affect what they can safely disclose.",
        choices: [
          {
            label: "Clarify privacy",
            text: "Explain confidentiality, portal access, billing limits, and documentation choices.",
            next: "risk",
            feedback: "<strong>Strong choice.</strong> Privacy is not administrative background here. It can affect safety, disclosure, and treatment engagement."
          },
          {
            label: "Ask parent to join",
            text: "Invite the parent into the next session to improve communication.",
            next: "end_privacy",
            feedback: "<strong>Slow down.</strong> Family involvement may help later, but only after assessing safety, consent, and the risk of outing or retaliation."
          },
          {
            label: "Continue assessment",
            text: "Move into suicide risk assessment and supports while noting privacy concerns.",
            next: "risk",
            feedback: "<strong>Reasonable path.</strong> Risk and privacy can be assessed together. Do not leave portal access unresolved."
          }
        ]
      },
      risk: {
        step: 3,
        label: "Risk and protection",
        question: "How would you complete this part of the assessment?",
        context: "The clinical picture now includes anxiety/depression, misgendering, avoidance, possible family rejection, privacy concerns, and self-silencing.",
        choices: [
          {
            label: "Assess both risk and protection",
            text: "Ask about suicide risk, immediate safety, chosen supports, and affirming spaces.",
            next: "end_strong",
            feedback: "<strong>Strong clinical branch.</strong> This connects symptoms to minority stress, safety, privacy, and protective affirmation."
          },
          {
            label: "Focus only on suicide risk",
            text: "Complete a suicide screen, then return to general anxiety treatment.",
            next: "end_partial",
            feedback: "<strong>Partly right.</strong> Suicide risk is essential, but the plan is incomplete without supports, privacy, family context, and affirmation."
          },
          {
            label: "Focus on coping skills",
            text: "Teach grounding skills and encourage the client to speak up at home.",
            next: "end_unsafe",
            feedback: "<strong>Not enough yet.</strong> Coping skills can help, but encouraging disclosure or confrontation before assessing safety may increase risk."
          }
        ]
      },
      end_strong: {
        step: 3,
        ending: true,
        label: "Branch complete",
        question: "Strong path: risk assessment plus protective affirmation",
        context: "You connected the client's symptoms to identity-based stress, family response, confidentiality, safety, and support.",
        feedback: "<strong>Clinical takeaway:</strong> Affirming care is not separate from risk assessment. It helps the clinician understand what is creating harm and what may create protection."
      },
      end_partial: {
        step: 3,
        ending: true,
        label: "Branch complete",
        question: "Partial path: risk identified, context still thin",
        context: "You included suicide risk, which matters, but the assessment still needs privacy, family context, identity stress, and protective supports.",
        feedback: "<strong>Clinical takeaway:</strong> A risk screen is necessary, but it is not the whole assessment. Ask what makes risk worse and what makes safety more possible."
      },
      end_unsafe: {
        step: 3,
        ending: true,
        label: "Branch complete",
        question: "Risky path: skills before safety",
        context: "Coping skills are useful, but they should not replace safety assessment or pressure the client to confront unsafe people.",
        feedback: "<strong>Clinical takeaway:</strong> Before recommending disclosure, confrontation, or family conversation, assess privacy, safety, supports, and possible retaliation."
      },
      end_minimize: {
        step: 3,
        ending: true,
        label: "Branch complete",
        question: "Minimizing path: key clues were missed",
        context: "The assessment treated self-silencing as ordinary conflict rather than a possible response to identity-based harm.",
        feedback: "<strong>Clinical takeaway:</strong> Avoid minimizing family rejection, misgendering, or silence. These may directly affect safety, depression, suicidality, and access to support."
      },
      end_privacy: {
        step: 3,
        ending: true,
        label: "Branch complete",
        question: "Privacy risk: family involvement came too soon",
        context: "Family involvement may be helpful only after the client gives informed consent and safety has been assessed.",
        feedback: "<strong>Clinical takeaway:</strong> When identity, billing, portals, and family access overlap, confidentiality planning is clinical care."
      }
    };

    var state = {
      current: "start",
      history: [],
      labels: [],
      selectedChoice: null,
      pendingNext: null
    };

    var step = root.querySelector("[data-branch-step]");
    var path = root.querySelector("[data-branch-path]");
    var progress = root.querySelector("[data-branch-progress]");
    var label = root.querySelector("[data-branch-label]");
    var question = root.querySelector("[data-branch-question]");
    var context = root.querySelector("[data-branch-context]");
    var choices = root.querySelector("[data-branch-choices]");
    var feedback = root.querySelector("[data-branch-feedback]");
    var submit = root.querySelector("[data-branch-submit]");
    var next = root.querySelector("[data-branch-next]");
    var back = root.querySelector("[data-branch-back]");

    function setSubmitEnabled(enabled){
      if(submit) submit.setAttribute("aria-disabled", enabled ? "false" : "true");
    }

    function setNextEnabled(enabled){
      if(next) next.setAttribute("aria-disabled", enabled ? "false" : "true");
    }

    function setText(el, text){
      if(el) el.textContent = text;
    }

    function render(){
      var node = nodes[state.current];
      if(!node) return;

      state.selectedChoice = null;
      state.pendingNext = null;
      setText(step, node.ending ? "Summary" : "Question " + node.step + " of 3");
      setText(path, state.labels.length ? state.labels.join(" / ") : "Start");
      if(progress) progress.style.width = ((node.ending ? 3 : node.step) / 3 * 100) + "%";
      setText(label, node.label);
      setText(question, node.question);
      setText(context, node.context);
      choices.innerHTML = "";
      feedback.classList.remove("is-visible");
      feedback.innerHTML = "";
      setSubmitEnabled(false);
      setNextEnabled(false);

      if(node.ending){
        feedback.innerHTML = node.feedback;
        feedback.classList.add("is-visible");
        setSubmitEnabled(false);
        setNextEnabled(false);
        back.setAttribute("aria-disabled", state.history.length ? "false" : "true");
        return;
      }

      node.choices.forEach(function(choice, index){
        var el = document.createElement("div");
        el.className = "cex-branch-choice";
        el.setAttribute("role", "button");
        el.setAttribute("tabindex", "0");
        el.setAttribute("data-choice-index", String(index));
        el.setAttribute("aria-pressed", "false");
        el.innerHTML = "<span>" + choice.label + "</span>" + choice.text;
        choices.appendChild(el);
      });

      back.setAttribute("aria-disabled", state.history.length ? "false" : "true");
    }

    function choose(choiceEl){
      var node = nodes[state.current];
      if(!node || node.ending) return;

      var selected = node.choices[Number(choiceEl.getAttribute("data-choice-index"))];
      if(!selected) return;

      Array.prototype.forEach.call(root.querySelectorAll(".cex-branch-choice"), function(el){
        el.classList.remove("is-selected");
        el.setAttribute("aria-pressed", "false");
      });
      choiceEl.classList.add("is-selected");
      choiceEl.setAttribute("aria-pressed", "true");
      state.selectedChoice = selected;
      state.pendingNext = null;
      setSubmitEnabled(true);
      setNextEnabled(false);

      feedback.classList.remove("is-visible");
      feedback.innerHTML = "";
    }

    function submitChoice(){
      var selected = state.selectedChoice;
      if(!selected) {
        feedback.innerHTML = "<strong>Select an answer first.</strong> Choose the clinical move you would make, then submit your answer.";
        feedback.classList.add("is-visible");
        return;
      }

      feedback.innerHTML = selected.feedback;
      feedback.classList.add("is-visible");
      setSubmitEnabled(false);
      state.pendingNext = selected;
      setNextEnabled(true);
    }

    function goNext(){
      var selected = state.pendingNext;
      if(!selected) return;

      state.history.push(state.current);
      state.labels.push(selected.label);
      state.current = selected.next;
      state.selectedChoice = null;
      state.pendingNext = null;
      render();
    }

    function goBack(){
      if(!state.history.length) return;
      state.current = state.history.pop();
      state.labels.pop();
      state.selectedChoice = null;
      state.pendingNext = null;
      render();
    }

    function reset(){
      state.current = "start";
      state.history = [];
      state.labels = [];
      state.selectedChoice = null;
      state.pendingNext = null;
      render();
    }

    function closestFrom(target, selector){
      while(target && target !== document){
        if(target.matches && target.matches(selector)) return target;
        target = target.parentNode;
      }
      return null;
    }

    function activate(event){
      var target = event.target && event.target.nodeType === 1 ? event.target : event.target.parentElement;
      var choice = closestFrom(target, ".cex-branch-choice");
      var submitBtn = closestFrom(target, "[data-branch-submit]");
      var nextBtn = closestFrom(target, "[data-branch-next]");
      var backBtn = closestFrom(target, "[data-branch-back]");
      var resetBtn = closestFrom(target, "[data-branch-reset]");
      var control = choice || submitBtn || nextBtn || backBtn || resetBtn;

      if(!control || !root.contains(control)) return;

      event.preventDefault();
      event.stopPropagation();

      if(choice){
        choose(choice);
        return;
      }
      if(submitBtn){
        submitChoice();
        return;
      }
      if(nextBtn){
        goNext();
        return;
      }
      if(backBtn){
        goBack();
        return;
      }
      if(resetBtn){
        reset();
      }
    }

    root.addEventListener("click", activate, true);
    root.addEventListener("keydown", function(event){
      if(event.key !== "Enter" && event.key !== " ") return;
      activate(event);
    }, true);

    render();
  }

  if(document.readyState === "loading"){
    document.addEventListener("DOMContentLoaded", initBranchingTree1932);
  } else {
    initBranchingTree1932();
  }

  window.addEventListener("load", initBranchingTree1932);
  setTimeout(initBranchingTree1932, 500);
  setTimeout(initBranchingTree1932, 1500);
})();
</script>
