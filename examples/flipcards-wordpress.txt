<script>
(function(){
  function initAuditFlipCardsFixed1932(){
    var root = document.getElementById("ce-audit-flipcards-fixed-1932");
    if(!root || root.dataset.ceafReady === "true") return;
    root.dataset.ceafReady = "true";

    root.addEventListener("click", function(event){
      var card = event.target.closest(".ceaf-card");
      if(!card || !root.contains(card)) return;
      event.preventDefault();
      event.stopPropagation();
      var flipped = card.classList.toggle("is-flipped");
      card.setAttribute("aria-pressed", flipped ? "true" : "false");
    }, true);
  }

  if(document.readyState === "loading"){
    document.addEventListener("DOMContentLoaded", initAuditFlipCardsFixed1932);
  } else {
    initAuditFlipCardsFixed1932();
  }

  window.addEventListener("load", initAuditFlipCardsFixed1932);
  setTimeout(initAuditFlipCardsFixed1932, 500);
  setTimeout(initAuditFlipCardsFixed1932, 1500);
})();
</script>
